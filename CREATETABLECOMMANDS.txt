CREATE TABLE User ( 
 GT_IC INT PRIMARY KEY, 
 Password VARCHAR(30) NOT NULL,
) 
 
CREATE TABLE Administrator ( 
 Password VARCHAR(30) NOT NULL,
 FOREIGN KEY (GT_IC) REFERENCES User(GT_IC)
) 
 
CREATE TABLE Student (
 Email VARCHAR(50) PRIMARY KEY,
 Name VARCHAR(30) NOT NULL,
 Password VARCHAR(30) NOT NULL,
 FOREIGN KEY (GT_IC) REFERENCES User(GT_IC)
)

CREATE TABLE Professor (
 Password VARCHAR(30) NOT NULL,
 FOREIGN KEY (GT_IC) REFERENCES User(GT_IC)
)

CREATE TABLE grad( 
 Name VARCHAR(30) NOT NULL,
 Password VARCHAR(30) NOT NULL,
 FOREIGN KEY (Email) REFERENCES Student (GT_IC),
 FOREIGN KEY (GT_IC) REFERENCES User(GT_IC)
) 

CREATE TABLE Undergrad( 
 Name VARCHAR(30) NOT NULL,
 Password VARCHAR(30) NOT NULL,
 FOREIGN KEY (Email) REFERENCES Student (GT_IC),
 FOREIGN KEY (GT_IC) REFERENCES User(GT_IC)
) 

CREATE TABLE Tutor(
 GPA DECIMAL(4,2) NOT NULL,
 Phone INT NOT NULL,
 Email VARCHAR(50) PRIMARY KEY,
 Name VARCHAR(30) NOT NULL,
 Password VARCHAR(30) NOT NULL,
 FOREIGN KEY (GT__IC) REFERENCES User(GT_IC)
)

CREATE TABLE Rating(
 Num_Evaluation DECIMAL(4,2) NOT NULL,
 Desc Evaluation VARCHAR(300) NOT NULL,
 Semester INT NOT NULL,
 FOREIGN KEY (GT_IC_TUTOR) REFERENCES Tutor(GT_IC),
 FOREIGN KEY (GT_IC_STUDENT) REFERENCES Undergrad(GT_IC)
)

CREATE TABLE Course(
 PRIMARY KEY (School, Number) NOT NULL,
)

CREATE TABLE TutorCourse(
 GTA VARCHAR(10) NOT NULL,
 FOREIGN KEY (TutorGT_ID) REFERENCES TutorTimeSlot(GT_ID),
 FOREIGN KEY (Course Key) REFERENCES Course(CourseKey)
)

CREATE TABLE HireTutor(
 FOREIGN KEY (UndergradGT_ID) REFERENCES Undergrad(GT_ID),
 FOREIGN KEY (CourseKey) REFERENCES Course(CourseKey),
 PRIMARY KEY (Slot Key) NOT NULL
)
 
CREATE TABLE TutorTimeSlot(
 Time VARCHAR(4) NOT NULL,
 Weekday VARCHAR(10) NOT NULL,
 Semester INT NOT NULL,
 PRIMARY KEY (TutorGT_ID) NOT NULL,
 FOREIGN KEY (Tutor) REFERENCES Tutor(GT_IC)
)

CREATE TABLE Recommendation(
 Num_Evaluation DECIMAL(4,2) NOT NULL,
 Desc_Evaluation VARCHAR(300) NOT NULL,
 FOREIGN KEY (TutorGT_ID) REFERENCES Tutor(GT_ID),
 FOREIGN KEY (ProfessorGT_ID) REFERENCES Professor(GT_ID)
)
